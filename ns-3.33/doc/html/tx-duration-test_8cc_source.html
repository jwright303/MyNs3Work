123 bool
124 TxDurationTest::CheckPayloadDuration (uint32_t size, WifiMode payloadMode, uint16_t channelWidth, uint16_t guardInterval, WifiPreamble preamble, Time knownDuration)
125 {
126  WifiTxVector txVector;
127  txVector.SetMode (payloadMode);
128  txVector.SetPreambleType (preamble);
129  txVector.SetChannelWidth (channelWidth);
130  txVector.SetGuardInterval (guardInterval);
131  txVector.SetNss (1);
132  txVector.SetStbc (0);
133  txVector.SetNess (0);
134  WifiPhyBand band = WIFI_PHY_BAND_2_4GHZ;
135  Ptr<YansWifiPhy> phy = CreateObject<YansWifiPhy> ();
136  if (payloadMode.GetModulationClass () == WIFI_MOD_CLASS_OFDM
137  || payloadMode.GetModulationClass () == WIFI_MOD_CLASS_HT
138  || payloadMode.GetModulationClass () == WIFI_MOD_CLASS_VHT
139  || payloadMode.GetModulationClass () == WIFI_MOD_CLASS_HE)
140  {
141  band = WIFI_PHY_BAND_5GHZ;
142  }
143  Time calculatedDuration = phy->GetPayloadDuration (size, txVector, band);
144  if (calculatedDuration != knownDuration)
145  {
146  std::cerr << "size=" << size
147  << " mode=" << payloadMode
148  << " channelWidth=" << channelWidth
149  << " guardInterval=" << guardInterval
150  << " datarate=" << payloadMode.GetDataRate (channelWidth, guardInterval, 1)
151  << " known=" << knownDuration
152  << " calculated=" << calculatedDuration
153  << std::endl;
154  return false;
155  }
156  if (payloadMode.GetModulationClass () == WIFI_MOD_CLASS_HT || payloadMode.GetModulationClass () == WIFI_MOD_CLASS_HE)
157  {
158  //Durations vary depending on frequency; test also 2.4 GHz (bug 1971)
159  band = WIFI_PHY_BAND_2_4GHZ;
160  calculatedDuration = phy->GetPayloadDuration (size, txVector, band);
161  knownDuration += MicroSeconds (6);
162  if (calculatedDuration != knownDuration)
163  {
164  std::cerr << "size=" << size
165  << " mode=" << payloadMode
166  << " channelWidth=" << channelWidth
167  << " guardInterval=" << guardInterval
168  << " datarate=" << payloadMode.GetDataRate (channelWidth, guardInterval, 1)
169  << " known=" << knownDuration
170  << " calculated=" << calculatedDuration
171  << std::endl;
172  return false;
173  }
174  }
175  return true;
176 }

241 bool
242 TxDurationTest::CheckHeMuTxDuration (std::list<uint32_t> sizes, std::list<HeMuUserInfo> userInfos,
243  uint16_t channelWidth, uint16_t guardInterval,
244  Time knownDuration)
245 {
246  NS_ASSERT (sizes.size () == userInfos.size () && sizes.size () > 1);
247  NS_ABORT_MSG_IF (channelWidth < std::accumulate (std::begin (userInfos), std::end (userInfos), 0,
248  [](const uint16_t prevBw, const HeMuUserInfo &info)
249  { return prevBw + HeRu::GetBandwidth (info.ru.ruType); }),
250  "Cannot accommodate all the RUs in the provided band"); //MU-MIMO (for which allocations use the same RU) is not supported
251  WifiTxVector txVector;
252  txVector.SetPreambleType (WIFI_PREAMBLE_HE_MU);
253  txVector.SetChannelWidth (channelWidth);
254  txVector.SetGuardInterval (guardInterval);
255  txVector.SetStbc (0);
256  txVector.SetNess (0);
257  std::list<uint16_t> staIds;
258  uint16_t staId = 1;
259  for (const auto & userInfo : userInfos)
260  {
261  txVector.SetHeMuUserInfo (staId, userInfo);
262  staIds.push_back (staId++);
263  }
264  Ptr<YansWifiPhy> phy = CreateObject<YansWifiPhy> ();
265  std::list<WifiPhyBand> testedBands {WIFI_PHY_BAND_5GHZ, WIFI_PHY_BAND_2_4GHZ}; //Durations vary depending on frequency; test also 2.4 GHz (bug 1971)
266  for (auto & testedBand : testedBands)
267  {
268  if (testedBand == WIFI_PHY_BAND_2_4GHZ)
269  {
270  knownDuration += MicroSeconds (6);
271  }
272  Time calculatedDuration = NanoSeconds (0);
273  uint32_t longuestSize = 0;
274  auto iterStaId = staIds.begin ();
275  for (auto & size : sizes)
276  {
277  Time ppduDurationForSta = phy->CalculateTxDuration (size, txVector, testedBand, *iterStaId);
278  if (ppduDurationForSta > calculatedDuration)
279  {
280  calculatedDuration = ppduDurationForSta;
281  staId = *iterStaId;
282  longuestSize = size;
283  }
284  ++iterStaId;
285  }
286  Time calculatedDurationUsingList = CalculateTxDurationUsingList (sizes, staIds, txVector, testedBand);
287  if (calculatedDuration != knownDuration || calculatedDuration != calculatedDurationUsingList)
288  {
289  std::cerr << "size=" << longuestSize
290  << " band=" << testedBand
291  << " staId=" << staId
292  << " nss=" << +txVector.GetNss (staId)
293  << " mode=" << txVector.GetMode (staId)
294  << " channelWidth=" << channelWidth
295  << " guardInterval=" << guardInterval
296  << " datarate=" << txVector.GetMode (staId).GetDataRate (channelWidth, guardInterval, txVector.GetNss (staId))
297  << " known=" << knownDuration
298  << " calculated=" << calculatedDuration
299  << " calculatedUsingList=" << calculatedDurationUsingList
300  << std::endl;
301  return false;
302  }
303  }
304  return true;
305 }

307 Time
308 TxDurationTest::CalculateTxDurationUsingList (std::list<uint32_t> sizes, std::list<uint16_t> staIds,
309  WifiTxVector txVector, WifiPhyBand band)
310 {
311  NS_ASSERT (sizes.size () == staIds.size ());
312  WifiConstPsduMap psduMap;
313  auto itStaId = staIds.begin ();
314  WifiMacHeader hdr;
315  hdr.SetType (WIFI_MAC_CTL_ACK); //so that size may not be empty while being as short as possible
316  for (auto & size : sizes)
317  {
318  // MAC header and FCS are to deduce from size
319  psduMap[*itStaId++] = Create<WifiPsdu> (Create<Packet> (size - hdr.GetSerializedSize () - 4), hdr);
320  }
321  return WifiPhy::CalculateTxDuration (psduMap, txVector, band);
322 }

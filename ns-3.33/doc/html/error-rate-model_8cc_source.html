38 double
39 ErrorRateModel::CalculateSnr (WifiTxVector txVector, double ber) const
40 {
41  //This is a very simple binary search.
42  double low, high, precision;
43  low = 1e-25;
44  high = 1e25;
45  precision = 2e-12;
46  while (high - low > precision)
47  {
48  NS_ASSERT (high >= low);
49  double middle = low + (high - low) / 2;
50  if ((1 - GetChunkSuccessRate (txVector.GetMode (), txVector, middle, 1)) > ber)
51  {
52  low = middle;
53  }
54  else
55  {
56  high = middle;
57  }
58  }
59  return low;
60 }

62 double
63 ErrorRateModel::GetChunkSuccessRate (WifiMode mode, WifiTxVector txVector, double snr, uint64_t nbits) const
64 {
65  if (mode.GetModulationClass () == WIFI_MOD_CLASS_DSSS || mode.GetModulationClass () == WIFI_MOD_CLASS_HR_DSSS)
66  {
67  switch (mode.GetDataRate (22, 0, 1))
68  {
69  case 1000000:
70  return DsssErrorRateModel::GetDsssDbpskSuccessRate (snr, nbits);
71  case 2000000:
72  return DsssErrorRateModel::GetDsssDqpskSuccessRate (snr, nbits);
73  case 5500000:
74  return DsssErrorRateModel::GetDsssDqpskCck5_5SuccessRate (snr, nbits);
75  case 11000000:
76  return DsssErrorRateModel::GetDsssDqpskCck11SuccessRate (snr, nbits);
77  default:
78  NS_ASSERT ("undefined DSSS/HR-DSSS datarate");
79  }
80  }
81  else
82  {
83  return DoGetChunkSuccessRate (mode, txVector, snr, nbits);
84  }
85  return 0;
86 }

200 uint32_t
201 FqCoDelQueueDisc::SetAssociativeHash (uint32_t flowHash)
202 {
203  NS_LOG_FUNCTION (this << flowHash);
204 
205  uint32_t h = (flowHash % m_flows);
206  uint32_t innerHash = h % m_setWays;
207  uint32_t outerHash = h - innerHash;
208 
209  for (uint32_t i = outerHash; i < outerHash + m_setWays; i++)
210  {
211  auto it = m_flowsIndices.find (i);
212 
213  if (it == m_flowsIndices.end ()
214  || (m_tags.find (i) != m_tags.end () && m_tags[i] == flowHash)
215  || StaticCast<FqCoDelFlow> (GetQueueDiscClass (it->second))->GetStatus () == FqCoDelFlow::INACTIVE)
216  {
217  // this queue has not been created yet or is associated with this flow
218  // or is inactive, hence we can use it
219  m_tags[i] = flowHash;
220  return i;
221  }
222  }
223 
224  // all the queues of the set are used. Use the first queue of the set
225  m_tags[outerHash] = flowHash;
226  return outerHash;
227 }

229 bool
230 FqCoDelQueueDisc::DoEnqueue (Ptr<QueueDiscItem> item)
231 {
232  NS_LOG_FUNCTION (this << item);
233 
234  uint32_t flowHash, h;
235 
236  if (GetNPacketFilters () == 0)
237  {
238  flowHash = item->Hash (m_perturbation);
239  }
240  else
241  {
242  int32_t ret = Classify (item);
243 
244  if (ret != PacketFilter::PF_NO_MATCH)
245  {
246  flowHash = static_cast<uint32_t> (ret);
247  }
248  else
249  {
250  NS_LOG_ERROR ("No filter has been able to classify this packet, drop it.");
251  DropBeforeEnqueue (item, UNCLASSIFIED_DROP);
252  return false;
253  }
254  }
255 
256  if (m_enableSetAssociativeHash)
257  {
258  h = SetAssociativeHash (flowHash);
259  }
260  else
261  {
262  h = flowHash % m_flows;
263  }
264 
265  Ptr<FqCoDelFlow> flow;
266  if (m_flowsIndices.find (h) == m_flowsIndices.end ())
267  {
268  NS_LOG_DEBUG ("Creating a new flow queue with index " << h);
269  flow = m_flowFactory.Create<FqCoDelFlow> ();
270  Ptr<QueueDisc> qd = m_queueDiscFactory.Create<QueueDisc> ();
271  // If CoDel, Set values of CoDelQueueDisc to match this QueueDisc
272  Ptr<CoDelQueueDisc> codel = qd->GetObject<CoDelQueueDisc> ();
273  if (codel)
274  {
275  codel->SetAttribute ("UseEcn", BooleanValue (m_useEcn));
276  codel->SetAttribute ("CeThreshold", TimeValue (m_ceThreshold));
277  codel->SetAttribute ("UseL4s", BooleanValue (m_useL4s));
278  }
279  qd->Initialize ();
280  flow->SetQueueDisc (qd);
281  flow->SetIndex (h);
282  AddQueueDiscClass (flow);
283 
284  m_flowsIndices[h] = GetNQueueDiscClasses () - 1;
285  }
286  else
287  {
288  flow = StaticCast<FqCoDelFlow> (GetQueueDiscClass (m_flowsIndices[h]));
289  }
290 
291  if (flow->GetStatus () == FqCoDelFlow::INACTIVE)
292  {
293  flow->SetStatus (FqCoDelFlow::NEW_FLOW);
294  flow->SetDeficit (m_quantum);
295  m_newFlows.push_back (flow);
296  }
297 
298  flow->GetQueueDisc ()->Enqueue (item);
299 
300  NS_LOG_DEBUG ("Packet enqueued into flow " << h << "; flow index " << m_flowsIndices[h]);
301 
302  if (GetCurrentSize () > GetMaxSize ())
303  {
304  NS_LOG_DEBUG ("Overload; enter FqCodelDrop ()");
305  FqCoDelDrop ();
306  }
307 
308  return true;
309 }

311 Ptr<QueueDiscItem>
312 FqCoDelQueueDisc::DoDequeue (void)
313 {
314  NS_LOG_FUNCTION (this);
315 
316  Ptr<FqCoDelFlow> flow;
317  Ptr<QueueDiscItem> item;
318 
319  do
320  {
321  bool found = false;
322 
323  while (!found && !m_newFlows.empty ())
324  {
325  flow = m_newFlows.front ();
326 
327  if (flow->GetDeficit () <= 0)
328  {
329  NS_LOG_DEBUG ("Increase deficit for new flow index " << flow->GetIndex ());
330  flow->IncreaseDeficit (m_quantum);
331  flow->SetStatus (FqCoDelFlow::OLD_FLOW);
332  m_oldFlows.push_back (flow);
333  m_newFlows.pop_front ();
334  }
335  else
336  {
337  NS_LOG_DEBUG ("Found a new flow " << flow->GetIndex () << " with positive deficit");
338  found = true;
339  }
340  }
341 
342  while (!found && !m_oldFlows.empty ())
343  {
344  flow = m_oldFlows.front ();
345 
346  if (flow->GetDeficit () <= 0)
347  {
348  NS_LOG_DEBUG ("Increase deficit for old flow index " << flow->GetIndex ());
349  flow->IncreaseDeficit (m_quantum);
350  m_oldFlows.push_back (flow);
351  m_oldFlows.pop_front ();
352  }
353  else
354  {
355  NS_LOG_DEBUG ("Found an old flow " << flow->GetIndex () << " with positive deficit");
356  found = true;
357  }
358  }
359 
360  if (!found)
361  {
362  NS_LOG_DEBUG ("No flow found to dequeue a packet");
363  return 0;
364  }
365 
366  item = flow->GetQueueDisc ()->Dequeue ();
367 
368  if (!item)
369  {
370  NS_LOG_DEBUG ("Could not get a packet from the selected flow queue");
371  if (!m_newFlows.empty ())
372  {
373  flow->SetStatus (FqCoDelFlow::OLD_FLOW);
374  m_oldFlows.push_back (flow);
375  m_newFlows.pop_front ();
376  }
377  else
378  {
379  flow->SetStatus (FqCoDelFlow::INACTIVE);
380  m_oldFlows.pop_front ();
381  }
382  }
383  else
384  {
385  NS_LOG_DEBUG ("Dequeued packet " << item->GetPacket ());
386  }
387  } while (item == 0);
388 
389  flow->IncreaseDeficit (item->GetSize () * -1);
390 
391  return item;
392 }

394 bool
395 FqCoDelQueueDisc::CheckConfig (void)
396 {
397  NS_LOG_FUNCTION (this);
398  if (GetNQueueDiscClasses () > 0)
399  {
400  NS_LOG_ERROR ("FqCoDelQueueDisc cannot have classes");
401  return false;
402  }
403 
404  if (GetNInternalQueues () > 0)
405  {
406  NS_LOG_ERROR ("FqCoDelQueueDisc cannot have internal queues");
407  return false;
408  }
409 
410  // we are at initialization time. If the user has not set a quantum value,
411  // set the quantum to the MTU of the device (if any)
412  if (!m_quantum)
413  {
414  Ptr<NetDeviceQueueInterface> ndqi = GetNetDeviceQueueInterface ();
415  Ptr<NetDevice> dev;
416  // if the NetDeviceQueueInterface object is aggregated to a
417  // NetDevice, get the MTU of such NetDevice
418  if (ndqi && (dev = ndqi->GetObject<NetDevice> ()))
419  {
420  m_quantum = dev->GetMtu ();
421  NS_LOG_DEBUG ("Setting the quantum to the MTU of the device: " << m_quantum);
422  }
423 
424  if (!m_quantum)
425  {
426  NS_LOG_ERROR ("The quantum parameter cannot be null");
427  return false;
428  }
429  }
430 
431  if (m_enableSetAssociativeHash && (m_flows % m_setWays != 0))
432  {
433  NS_LOG_ERROR ("The number of queues must be an integer multiple of the size "
434  "of the set of queues used by set associative hash");
435  return false;
436  }
437 
438  if (m_useL4s)
439  {
440  NS_ABORT_MSG_IF (m_ceThreshold == Time::Max(), "CE threshold not set");
441  if (m_useEcn == false)
442  {
443  NS_LOG_WARN ("Enabling ECN as L4S mode is enabled");
444  }
445  }
446  return true;
447 }

462 uint32_t
463 FqCoDelQueueDisc::FqCoDelDrop (void)
464 {
465  NS_LOG_FUNCTION (this);
466 
467  uint32_t maxBacklog = 0, index = 0;
468  Ptr<QueueDisc> qd;
469 
470  /* Queue is full! Find the fat flow and drop packet(s) from it */
471  for (uint32_t i = 0; i < GetNQueueDiscClasses (); i++)
472  {
473  qd = GetQueueDiscClass (i)->GetQueueDisc ();
474  uint32_t bytes = qd->GetNBytes ();
475  if (bytes > maxBacklog)
476  {
477  maxBacklog = bytes;
478  index = i;
479  }
480  }
481 
482  /* Our goal is to drop half of this fat flow backlog */
483  uint32_t len = 0, count = 0, threshold = maxBacklog >> 1;
484  qd = GetQueueDiscClass (index)->GetQueueDisc ();
485  Ptr<QueueDiscItem> item;
486 
487  do
488  {
489  NS_LOG_DEBUG ("Drop packet (overflow); count: " << count << " len: " << len << " threshold: " << threshold);
490  item = qd->GetInternalQueue (0)->Dequeue ();
491  DropAfterDequeue (item, OVERLIMIT_DROP);
492  len += item->GetSize ();
493  } while (++count < m_dropBatchSize && len < threshold);
494 
495  return index;
496 }
